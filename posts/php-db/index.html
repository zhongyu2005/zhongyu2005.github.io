public function getSqlite($path)
{
    if (empty($this->dbs[$path])) {
        $this->dbs[$path] = new \SQLite3($path);
    }
    return $this->dbs[$path];
}

public function dbExec($db, $sql,$bindParam)
{
    $sm=$db->prepare($sql);
    if(!empty($bindParam)){
        foreach ($bindParam as $key=>$val) {
            $sm->bindValue($key, $val);
        }
    }
    return $sm->execute();
    // $db->lastErrorMsg()
}

public function dbQueryOne($db, $sql,$bindParam)
{
    $sm=$db->prepare($sql);
    if(!empty($bindParam)){
        foreach ($bindParam as $key=>$val) {
            $sm->bindValue($key, $val);
        }
    }
    $res = $sm->execute();
    $row = $res->fetchArray(SQLITE3_ASSOC);
    return $row;
}

public function dbQueryAll($db, $sql,$bindParam)
{
    $sm=$db->prepare($sql);
    if(!empty($bindParam)){
        foreach ($bindParam as $key=>$val) {
            $sm->bindValue($key, $val);
        }
    }
    $res = $sm->execute();
    $list = [];
    while ($row = $res->fetchArray(SQLITE3_ASSOC)) {
        $list[] = $row;
    }
    return $list;
}

function mysqlInsert($tab,array $set){
    $sql="INSERT INTO %s (%s) values(%s) ";
    $key=$pre=$val=[];
    foreach ($set as $k => $v) {
        $key[]='`'.$k.'`';
        $pre[]='?';
        $val[]=$v;
    }
    $sql=sprintf($sql, $tab, implode(',', $key), implode(',', $pre));
    if(empty($db)){
        return [$sql,$val];
    }
    $sth=$db->prepare($sql);
    $r=$sth->execute($val);
    if($r){
        return $this->db->lastInsertId();
    }
    return $r;
}


function mysqlUpdate($tab,$where,array $set){
    $sql='UPDATE `'.$tab.'` SET ';
    $key=$val=[];
    foreach ($set as $k => $v) {
        $key[]='`'.$k.'`=?';
        $val[]=$v;
    }
    $sql.=implode(',', $key) .' WHERE '.$where;
    if (empty($db)) {
        return [$sql,$val];
    }
    $sth=$db->prepare($sql);
    $r=$sth->execute($val);
    if ($r) {
        //返回受影响的行数
        return $sth->rowCount();
    }
    //返回true or false
    return $r;
}

function mysqlSelect($select,$tab,$where){
    $sql="SELECT {$select} FROM {$tab} WHERE {$where}";
