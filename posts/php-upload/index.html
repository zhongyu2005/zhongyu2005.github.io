upload php


$file = $_FILES['file'];
if (empty($file)) {
    return log("未获取到上传数据", 1);
}
$maxSize = 1024 * 1024 * 3;//3m
if ($file['size'] > $maxSize) {
    return log("上传数据大小超过限制", 1);
}
if (!in_array($file['type'], ['image/jpeg', 'image/png', 'image/gif'])) {
    log("上传文件类型不正确", 1);
}
$suffix = pathinfo($file['name'], PATHINFO_EXTENSION);   //文件后缀
$suffix = strtolower($suffix);
if (!in_array($suffix, array('jpg', 'png', 'gif', 'jpeg'))) {
    return log("上传文件类型不正确", 1);
}
$path = __DIR__ . '/img';
$name = '/' . date("Ym") . '/' . md5(uniqid('dp', true)) . '.' . $suffix;
if (!is_dir(dirname($path . $name))) {
    mkdir(dirname($path . $name), 0775, true);
}
$f = move_uploaded_file($_FILES['file']['tmp_name'], $path . $name);
if (empty($f)) {
    return log("上传失败", 1);
}
return "success";
